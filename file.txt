#include <cstdio>
#include <iostream>


#include <netinet/in.h>
#include <arpa/inet.h>
#include <sys/socket.h>
#include <unistd.h>
#include <cstring>

template<class T, class... Args>
void Info(const T v, Args... args) {
    std::cout << "[INFO]: ";
    (std::cout << ... << args) << "\n";
}


int main(int argc, char** argv) {
    if (argc <= 1) {
        std::cout << "Arguments not enought, args: ip port message\n";
        return 0;
    }
    const char* ADDRESS = argv[1];
    const int PORT = atoi(argv[2]);
    const char* MESSAGE = argv[3];
    const int AF = AF_INET;

    int sfd;
    in_addr inAddr;
    sockaddr_in sockAddr;


    if (inet_pton(AF, ADDRESS, &inAddr) == -1) {
        std::perror("inet_pton");
        exit(-1);
    }
    sockAddr.sin_family = AF;
    sockAddr.sin_addr = inAddr;
    sockAddr.sin_port = htons(static_cast<uint16_t>(PORT));

    if ((sfd = socket(AF, SOCK_STREAM, 0)) == -1) {
        std::perror("socket");
        exit(-1);
    }
    
    if (connect(sfd, (sockaddr*)&sockAddr, sizeof(sockaddr)) == -1) {
        std::perror("connect");
        exit(-1);
    }

    if (write(sfd, MESSAGE, strlen(MESSAGE)) == -1) {
        std::perror("write");
        exit(-1);
    }

    char buffer[128];
    int len;
    if ((len = read(sfd, buffer, 128)) == -1) {
        std::perror("read");
        exit(-1);
    }

    std::cout.write(buffer, len) << "\n";

    close(sfd);
    
    return 0;
}
